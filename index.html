Temperature Converter

from js import document def convert(event): temp = float(document.getElementById("temp").value) choice = document.getElementById("choice").value if choice == "c_to_f": result = (temp * 9/5) + 32 else: result = (temp - 32) * 5/9 document.getElementById("result").innerText = f"Converted Temperature: {result:.2f}" document.getElementById("btn").addEventListener("click", convert)
